BUILD IMMEDIATE

AS

-- 6. Audit Logging & Reporting
-- Create Audit Table
CREATE TABLE audit_logs (
    log_id    NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    commander VARCHAR2(100),
    action    VARCHAR2(100),
    object    VARCHAR2(100),
    timestamp TIMESTAMP DEFAULT SYSTIMESTAMP
);

-- Procedure to generate disaster report and log the action
CREATE OR REPLACE PROCEDURE generate_disaster_report IS
BEGIN
    INSERT INTO audit_logs (commander, action, object, timestamp)
    VALUES ('SYSTEM', 'GENERATE_REPORT', 'GOVT_BRIEFING', SYSTIMESTAMP);

    COMMIT;
END;

-- 9. UTL_FILE package
CREATE OR REPLACE PROCEDURE generate_disaster_report IS
    v_file UTL_FILE.FILE_TYPE;
BEGIN
    -- Step 1: Write to the database table (for your UI Audit Logs)
    INSERT INTO audit_logs (commander, action, object, timestamp)
    VALUES ('SYSTEM', 'GENERATE_REPORT', 'GOVT_BRIEFING', SYSTIMESTAMP);

    -- Step 2: Write to the physical file in C:\ocrl_databas
    -- Note: 'LOG_DIR' must be a valid Oracle Directory object
    v_file := UTL_FILE.FOPEN('LOG_DIR', 'disaster_report.txt', 'W');
    
    UTL_FILE.PUT_LINE(v_file, '--- DISASTER MANAGEMENT SYSTEM OFFICIAL REPORT ---');
    UTL_FILE.PUT_LINE(v_file, 'Generated on: ' || TO_CHAR(SYSDATE, 'YYYY-MM-DD HH24:MI:SS'));
    UTL_FILE.PUT_LINE(v_file, 'Status: System Online and Audited.');
    
    UTL_FILE.FCLOSE(v_file);

    COMMIT;
EXCEPTION
    WHEN OTHERS THEN
        -- Safety check: Close the file if an error occurs while it's open
        IF UTL_FILE.IS_OPEN(v_file) THEN
            UTL_FILE.FCLOSE(v_file);
        END IF;
        RAISE;
END;
/



-- 10. Triggers

-- 1. Volunteer Audit Trigger
CREATE OR REPLACE TRIGGER trg_volunteer_audit
AFTER INSERT OR DELETE ON volunteers
FOR EACH ROW
BEGIN
    IF INSERTING THEN
        INSERT INTO audit_logs (commander, action, object, timestamp)
        VALUES ('SYSTEM', 'INSERT', 'Volunteer: ' || :NEW.name, SYSTIMESTAMP);
    ELSIF DELETING THEN
        INSERT INTO audit_logs (commander, action, object, timestamp)
        VALUES ('SYSTEM', 'DELETE', 'Volunteer: ' || :OLD.name, SYSTIMESTAMP);
    END IF;
END;
/

-- 2. Resource Audit Trigger
CREATE OR REPLACE TRIGGER trg_resource_audit
AFTER UPDATE ON resources
FOR EACH ROW
BEGIN
    INSERT INTO audit_logs (commander, action, object, timestamp)
    VALUES (
        'SYSTEM',
        'UPDATE',
        'Resource: ' || :NEW.name || ' (New Qty: ' || :NEW.current_stock || ' )',
        SYSTIMESTAMP
    );
END;
/
